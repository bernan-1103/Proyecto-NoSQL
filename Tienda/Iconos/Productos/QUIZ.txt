use QuizDoblePuntaje

Select m.DESCRIPCION,sum(f.TOTAL) Total,sum(f.IMPUESTO)Totalimp,sum(f.DESCUENTO)TotalDes
from TFACTURA f,TMONEDA m
where f.CO_MONEDA_FK=m.[COD_MONEDA  PK]
group by m.DESCRIPCION

Select v.NOMBRE, m.DESCRIPCION,sum(f.TOTAL) Total
from TFACTURA f,TMONEDA m,TVENDEDORES V,TCLIENTE C
where (f.CO_MONEDA_FK=m.[COD_MONEDA  PK] and V.[NO_VENDE
PK]=C.NO_VENDE_FK and C.[NO_CLIENTE 
PK]=f.NO_CLIENTE_FK) and V.NOMBRE='EDGAR' and f.FECHA>01/03/2018 and f.FECHA<17/032018
group by V.NOMBRE,m.DESCRIPCION


select * 
from TFACTURA

Select c.NOMBRE, m.DESCRIPCION,sum(f.TOTAL*f.TCAMBIO) TotalColones,sum(f.IMPUESTO*f.TCAMBIO)TotalimpColones
from TFACTURA f,TMONEDA m,TCLIENTE C
where (f.CO_MONEDA_FK=m.[COD_MONEDA  PK] and C.[NO_CLIENTE 
PK]=f.NO_CLIENTE_FK)
group by c.NOMBRE,m.DESCRIPCION
having Sum(f.total*f.TCAMBIO) >50000


Select p.DESCRIPCION, m.DESCRIPCION,sum(f.TOTAL) Total
from TFACTURA f,TMONEDA m,TCLIENTE C,TPROVINCIA P	
where (f.CO_MONEDA_FK=m.[COD_MONEDA  PK] and C.[NO_CLIENTE 
PK]=f.NO_CLIENTE_FK and p.[PROVINCIA PK]=c.PROVINCIA_FK)
group by p.DESCRIPCION,m.DESCRIPCION
order by Total desc


Select sum(f.TOTAL*f.TCAMBIO) TotalColones,sum(f.IMPUESTO*f.TCAMBIO)TotalimpColones,sum(f.DESCUENTO*f.TCAMBIO)TotalDesColones
from TFACTURA f,TMONEDA m,TCLIENTE C
where (f.CO_MONEDA_FK=m.[COD_MONEDA  PK] and C.[NO_CLIENTE 
PK]=f.NO_CLIENTE_FK)

Select p.DESCRIPCION,m.DESCRIPCION,pg.DESCRIPCION,sum(f.SUBTOTAL) TotalSub,sum(f.DESCUENTO)TotalDesc,sum(f.TOTAL)Total
from TFACTURA F,TPROVINCIA P,TMONEDA M,TPAGO PG,TCLIENTE C
where (f.CO_MONEDA_FK=m.[COD_MONEDA  PK] and f.T_PAGO_FK=pg.[T_PAGO PK] and c.PROVINCIA_FK=p.[PROVINCIA PK]and c.[NO_CLIENTE 
PK]=f.NO_CLIENTE_FK)
group by p.DESCRIPCION,m.DESCRIPCION,pg.DESCRIPCION
order by 1,2,3
